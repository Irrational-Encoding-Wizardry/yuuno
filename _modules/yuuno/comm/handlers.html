[docs]    def dispatch(self, comm: YuunoCommConnection, command: Command):
        name = command.command_name
        func = getattr(self, "on_%s"%name, lambda _, __:None)
        return func(comm, command)



[docs]    def on_update(self, comm: YuunoCommConnection, command: UpdateCommand):
        comm.send(UpdateCommand(self.extension.output_mirror.current_as_changeset(), id=command.id))



[docs]    def on_request_output(self, comm: YuunoCommConnection, command: OutputRequestCommand):
        clip = VapourSynthEnvironment.get_global_outputs()[command.output]
        wrapper = self.yuuno.wrap(clip)
        comm.send(OutputResponseCommand.from_clip(command.id, wrapper, command.settings))



[docs]    def on_request_frame(self, comm: YuunoCommConnection, command: FrameRequestCommand):
        clip = VapourSynthEnvironment.get_global_outputs()[command.output]
        wrapper = self.yuuno.wrap(clip)[command.frame]
        comm.send(FrameResponseCommand(command.id, wrapper.to_pil()))
